## 3. 性能测试脚本

### 3.1 帧率监控脚本

```typescript
// performance/fps-monitor.ts

interface FPSStats {
  current: number;
  average: number;
  min: number;
  max: number;
  samples: number[];
}

class FPSMonitor {
  private stats: FPSStats = {
    current: 0,
    average: 0,
    min: Infinity,
    max: 0,
    samples: []
  };
  
  private lastTime: number = 0;
  private frameCount: number = 0;
  private readonly SAMPLE_SIZE = 60; // 采样 60 帧

  start(): void {
    this.lastTime = performance.now();
    this.requestFrame();
  }

  private requestFrame(): void {
    requestAnimationFrame(() => {
      const now = performance.now();
      const delta = now - this.lastTime;
      
      if (delta >= 1000) {
        // 计算 FPS
        const fps = Math.round((this.frameCount * 1000) / delta);
        this.updateStats(fps);
        
        // 重置计数器
        this.frameCount = 0;
        this.lastTime = now;
      }
      
      this.frameCount++;
      this.requestFrame();
    });
  }

  private updateStats(fps: number): void {
    this.stats.current = fps;
    this.stats.samples.push(fps);
    
    if (this.stats.samples.length > this.SAMPLE_SIZE) {
      this.stats.samples.shift();
    }
    
    this.stats.average = Math.round(
      this.stats.samples.reduce((a, b) => a + b, 0) / this.stats.samples.length
    );
    this.stats.min = Math.min(this.stats.min, fps);
    this.stats.max = Math.max(this.stats.max, fps);
  }

  getStats(): FPSStats {
    return { ...this.stats };
  }

  reset(): void {
    this.stats = {
      current: 0,
      average: 0,
      min: Infinity,
      max: 0,
      samples: []
    };
    this.frameCount = 0;
    this.lastTime = performance.now();
  }
}

// 使用示例
const fpsMonitor = new FPSMonitor();
fpsMonitor.start();

// 定期上报性能数据
setInterval(() => {
  const stats = fpsMonitor.getStats();
  reportPerformance('fps', stats);
  
  // 性能告警
  if (stats.average < 30) {
    reportWarning('FPS过低', stats);
  }
}, 5000);
```

### 3.2 内存监控脚本

```typescript
// performance/memory-monitor.ts

interface MemoryStats {
  usedJSHeapSize: number;
  totalJSHeapSize: number;
  jsHeapSizeLimit: number;
  timestamp: number;
}

class MemoryMonitor {
  private history: MemoryStats[] = [];
  private readonly MAX_HISTORY = 100;

  sample(): MemoryStats | null {
    // @ts-ignore - performance.memory 是非标准 API
    if (performance.memory) {
      // @ts-ignore
      const stats: MemoryStats = {
        usedJSHeapSize: performance.memory.usedJSHeapSize,
        totalJSHeapSize: performance.memory.totalJSHeapSize,
        jsHeapSizeLimit: performance.memory.jsHeapSizeLimit,
        timestamp: Date.now()
      };
      
      this.history.push(stats);
      if (this.history.length > this.MAX_HISTORY) {
        this.history.shift();
      }
      
      return stats;
    }
    return null;
  }

  getMemoryUsageMB(): number {
    const last = this.history[this.history.length - 1];
    return last ? last.usedJSHeapSize / 1024 / 1024 : 0;
  }

  checkMemoryLeak(): boolean {
    if (this.history.length < 10) return false;
    
    // 检测内存是否持续增长
    const recent = this.history.slice(-10);
    const first = recent[0].usedJSHeapSize;
    const last = recent[recent.length - 1].usedJSHeapSize;
    
    // 如果增长了 20% 以上，可能存在内存泄漏
    return (last - first) / first > 0.2;
  }

  getHistory(): MemoryStats[] {
    return [...this.history];
  }
}

// 使用示例
const memoryMonitor = new MemoryMonitor();

// 定期采样
setInterval(() => {
  const stats = memoryMonitor.sample();
  if (stats) {
    const mb = memoryMonitor.getMemoryUsageMB();
    reportPerformance('memory', { mb });
    
    // 内存告警
    if (mb > 200) {
      reportWarning('内存占用过高', { mb });
    }
    
    // 内存泄漏检测
    if (memoryMonitor.checkMemoryLeak()) {
      reportWarning('可能存在内存泄漏', memoryMonitor.getHistory());
    }
  }
}, 1000);
```

### 3.3 加载时间监控脚本

```typescript
// performance/load-time-monitor.ts

interface LoadTimeMetrics {
  coldStart: number; // 冷启动时间
  warmStart: number; // 热启动时间
  firstScreen: number; // 首屏时间
  levelLoad: Record<string, number>; // 各关卡加载时间
  resourceLoad: Record<string, number>; // 资源加载时间
}

class LoadTimeMonitor {
  private metrics: LoadTimeMetrics = {
    coldStart: 0,
    warmStart: 0,
    firstScreen: 0,
    levelLoad: {},
    resourceLoad: {}
  };

  private startTime: number = 0;
  private isColdStart: boolean = true;

  init(): void {
    this.startTime = Date.now();
    
    // 判断是否冷启动
    const launchInfo = wx.getLaunchOptionsSync?.();
    this.isColdStart = !launchInfo?.scene;
  }

  markFirstScreen(): void {
    const time = Date.now() - this.startTime;
    this.metrics.firstScreen = time;
    reportPerformance('firstScreen', time);
  }

  markLevelLoad(levelId: string): void {
    const start = Date.now();
    return () => {
      const time = Date.now() - start;
      this.metrics.levelLoad[levelId] = time;
      reportPerformance('levelLoad', { levelId, time });
    };
  }

  markResourceLoad(resourceId: string): void {
    const start = Date.now();
    return () => {
      const time = Date.now() - start;
      this.metrics.resourceLoad[resourceId] = time;
    };
  }

  getMetrics(): LoadTimeMetrics {
    return { ...this.metrics };
  }

  report(): void {
    // 上报完整性能数据
    reportPerformance('loadTime', this.metrics);
  }
}

// 使用示例
const loadMonitor = new LoadTimeMonitor();
loadMonitor.init();

// 首屏加载完成
wx.onShow(() => {
  loadMonitor.markFirstScreen();
});

// 关卡加载
const onLevelLoad = loadMonitor.markLevelLoad('level-1');
loadLevel('level-1').then(() => {
  onLevelLoad();
});
```

### 3.4 性能测试自动化脚本

```typescript
// performance/auto-test.ts

import { FPSMonitor } from './fps-monitor';
import { MemoryMonitor } from './memory-monitor';
import { LoadTimeMonitor } from './load-time-monitor';

interface PerformanceReport {
  timestamp: number;
  device: string;
  wechatVersion: string;
  fps: {
    average: number;
    min: number;
    max: number;
    pass: boolean;
  };
  memory: {
    average: number;
    peak: number;
    pass: boolean;
  };
  loadTime: {
    firstScreen: number;
    levelLoad: Record<string, number>;
    pass: boolean;
  };
  overall: 'PASS' | 'WARNING' | 'FAIL';
}

class PerformanceTestSuite {
  private fpsMonitor = new FPSMonitor();
  private memoryMonitor = new MemoryMonitor();
  private loadMonitor = new LoadTimeMonitor();
  private results: PerformanceReport[] = [];

  async runAllTests(): Promise<PerformanceReport> {
    this.loadMonitor.init();
    this.fpsMonitor.start();

    const report: PerformanceReport = {
      timestamp: Date.now(),
      device: this.getDeviceInfo(),
      wechatVersion: this.getWechatVersion(),
      fps: { average: 0, min: 0, max: 0, pass: true },
      memory: { average: 0, peak: 0, pass: true },
      loadTime: { firstScreen: 0, levelLoad: {}, pass: true },
      overall: 'PASS'
    };

    // 测试启动性能
    await this.testStartup(report);

    // 测试游戏过程性能
    await this.testGameplay(report);

    // 计算总体结果
    report.overall = this.calculateOverall(report);

    this.results.push(report);
    return report;
  }

  private async testStartup(report: PerformanceReport): Promise<void> {
    // 等待首屏
    await new Promise(resolve => setTimeout(resolve, 3000));
    report.loadTime.firstScreen = this.loadMonitor.getMetrics().firstScreen;
    
    // 检查首屏时间
    if (report.loadTime.firstScreen > 5000) {
      report.loadTime.pass = false;
      report.overall = 'FAIL';
    } else if (report.loadTime.firstScreen > 3000) {
      report.loadTime.pass = false;
    }
  }

  private async testGameplay(report: PerformanceReport): Promise<void> {
    // 测试 5 分钟
    const duration = 5 * 60 * 1000;
    const startTime = Date.now();

    while (Date.now() - startTime < duration) {
      // 采样内存
      this.memoryMonitor.sample();
      
      // 模拟游戏操作
      await this.simulateGameplay();
      
      await new Promise(resolve => setTimeout(resolve, 1000));
    }

    // 计算 FPS 统计
    const fpsStats = this.fpsMonitor.getStats();
    report.fps = {
      average: fpsStats.average,
      min: fpsStats.min,
      max: fpsStats.max,
      pass: fpsStats.average >= 30
    };

    // 计算内存统计
    const memoryHistory = this.memoryMonitor.getHistory();
    const memoryValues = memoryHistory.map(m => m.usedJSHeapSize / 1024 / 1024);
    report.memory = {
      average: memoryValues.reduce((a, b) => a + b, 0) / memoryValues.length,
      peak: Math.max(...memoryValues),
      pass: Math.max(...memoryValues) <= 300
    };
  }

  private async simulateGameplay(): Promise<void> {
    // 模拟用户操作
    // 实际测试中需要真实的用户交互
  }

  private calculateOverall(report: PerformanceReport): 'PASS' | 'WARNING' | 'FAIL' {
    if (!report.fps.pass || !report.memory.pass || !report.loadTime.pass) {
      return 'FAIL';
    }
    if (report.fps.average < 50 || report.memory.average > 150) {
      return 'WARNING';
    }
    return 'PASS';
  }

  private getDeviceInfo(): string {
    const info = wx.getSystemInfoSync();
    return `${info.brand} ${info.model}`;
  }

  private getWechatVersion(): string {
    return wx.getAccountInfoSync?.()?.miniProgram?.version || 'unknown';
  }

  getResults(): PerformanceReport[] {
    return [...this.results];
  }
}

export const performanceTestSuite = new PerformanceTestSuite();
```

## 5. 性能优化建议

### 5.1 渲染优化
- 使用精灵图减少绘制调用
- 合理设置渲染区域
- 避免过度绘制
- 使用离屏渲染缓存静态内容

### 5.2 内存优化
- 及时释放不用的资源
- 使用对象池复用对象
- 压缩图片资源
- 按需加载资源

### 5.3 代码优化
- 避免频繁创建对象
- 使用 Web Worker 处理复杂计算
- 优化物理引擎迭代次数
- 减少不必要的碰撞检测

### 5.4 资源优化
- 压缩音频文件
- 使用纹理图集
- 分包加载
- 资源预加载策略
